DOCUMENT -> BLOCKLIST
BLOCKLIST -> BLOCKLIST BLOCK
BLOCKLIST -> BLOCK
BLOCK -> HEADING
BLOCK -> PARAGRAPH
HEADING -> HASH TEXT NEWLINE
PARAGRAPH -> TEXT NEWLINE
TEXT -> TEXT ELEMENT
TEXT -> ELEMENT
ELEMENT -> BOLD
ELEMENT -> ITALICS
ELEMENT -> PLAIN_TEXT
BOLD -> DOUBLE_AST PLAIN_TEXT DOUBLE_AST
ITALICS -> ASTERISK PLAIN_TEXT ASTERISK
