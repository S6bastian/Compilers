DOCUMENT    -> BLOCKLIST
BLOCKLIST   -> BLOCKLIST BLOCK
BLOCKLIST   -> BLOCK

BLOCK       -> HEADING
BLOCK       -> PARAGRAPH

HEADING     -> # TEXT \n
PARAGRAPH   -> TEXT \n

TEXT        -> TEXT ELEMENT
TEXT        -> ELEMENT

ELEMENT     -> BOLD
ELEMENT     -> ITALICS
ELEMENT     -> PLAIN_TEXT

BOLD        -> ** PLAIN_TEXT **
ITALICS     -> * PLAIN_TEXT *
